// 0. μμ‹ μΈν’‹
const [N, M] = [4, 4];
const arr = [1, 1, 2, 2];
const newArr = [...new Set([...arr])].sort((a, b) => a - b);

const answerArr = [];

// 1. dfs
const dfs = (start, pickArr) => {
  if (pickArr.length === M) {
    answerArr.push(pickArr.join(" "));
    return;
  }

  for (let i = start; i < newArr.length; i++) {
    pickArr.push(newArr[i]);
    dfs(i, pickArr);
    pickArr.pop();
  }
};

dfs(0, []);

const answer = answerArr.join("\n");
console.log(answer);
