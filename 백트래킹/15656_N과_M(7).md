const dfs = pickArr => {
  if (pickArr.length === M) {

    return;

  for (let i = 0; i < N; i++) {
    pickArr.push(arr[i]);
    dfs(pickArr);
    pickArr.pop();
  }
};

dfs([]);

console.log(answer);
