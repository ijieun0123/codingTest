function dfs(pickArr = []) {
  if (pickArr.length === M) {
    result += pickArr.join(" ") + "\n";
    return;
  }

  for (let i = 1; i <= N; i++) {
    pickArr.push(i);
    dfs(pickArr);
    pickArr.pop();
  }
}
