```js
// 0. 인풋 예시
const [N, M] = [4, 2];
const arr = [9, 8, 7, 1].sort((a, b) => a - b);

let answer = "";

// 1. DFS 중복조합
const dfs = (start, pickArr) => {
  if (pickArr.length === M) {
    answer += pickArr.join(" ") + "\n";
    return;
  }

  for (let i = start; i < N; i++) {
    pickArr.push(arr[i]);
    dfs(i, pickArr);
    pickArr.pop();
  }
};

dfs(0, []);

console.log(answer);
```
