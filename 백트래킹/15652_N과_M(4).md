function dfs(start, pickArr) {
  if (pickArr.length === M) {
    result += pickArr.join(" ") + "\n";
    return;
  }

  for (let i = start; i <= N; i++) {
    pickArr.push(i);
    dfs(i, pickArr);
    pickArr.pop();
  }
}
