const factorial = num => {
  if (num <= 1) return 1;
  return num * factorial(num - 1);
};

inputArr.forEach((arrEl, i) => {
  const [N, M] = arrEl;
  const count = Math.round(factorial(M) / (factorial(M - N) * factorial(N)));
  console.log(count);
});
