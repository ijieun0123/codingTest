  if (map.has(N)) {
    max = Math.max(max, map.get(N) + 1);
    map.set(N, map.get(N) + 1);
  } else map.set(N, 1);

const maxArr = [];
for (let [key, value] of map) {
  if (value === max) maxArr.push(key);
}

// 4. ë²”ìœ„
