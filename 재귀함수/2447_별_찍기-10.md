const fs = require("fs");
const input = Number(fs.readFileSync("/dev/stdin").toString().trim());

const star = n => {
  if (n === 1) {
    return ["*"];
  }
  const prevStar = star(n / 3);
  const top = prevStar.map(s => s.repeat(3));
  const mid = prevStar.map(s => s + " ".repeat(s.length) + s);

  return [...top, ...mid, ...top];
};
